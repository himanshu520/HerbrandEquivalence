396  void decreaseParentCnt(IDstruct *&ptr) {
397  if(ptr == nullptr) return;
398  assert(ptr->parentCnt > 0);
399 
400  ptr->parentCnt -= 1;
401  if(ptr->parentCnt > 0) return;
402 
403  // object pointed by ptr has no pointers to it,
404  // so it can now be destroyed
405 
406  // first check whether the object has some children
407  // (in the form of leftID and rightID), and if so
408  // first call decreaseParentCnt for them recursively
409  // (here we can also use `ptr->opSymbol != ""`
410  // or `ptr->left != nullptr` or `ptr->right != nullptr`)
411  if(ptr->leftID != nullptr and ptr->rightID != nullptr) {

710  void transferFunction(Partition &curPart, Partition const &prevPart,
711  Instruction &I) {
712 
713  // first copy the partition at predecessor instruction
714  // and then make required changes
715  copyPartition(curPart, prevPart);
716 
717  // changed stores the variable whose equivalence has
718  // been changed
719  Value *changed = nullptr;
720 
721  Value *left, *right;
722  IDstruct *leftID, *rightID, *IID, *newID;
723 
724  if(isa<LoadInst>(I)) {
725  left = I.getOperand(0);
726  leftID = curPart[indexCV[left]];
727  IID = curPart[indexCV[&I]];
728 
729  decreaseParentCnt(IID);
730  curPart[indexCV[&I]] = leftID;
731  increaseParentCnt(leftID);
732 
733  changed = &I;
734  } else if(isa<StoreInst>(I)) {
735  left = I.getOperand(0);
736  right = I.getOperand(1);
737  leftID = curPart[indexCV[left]];
738  rightID = curPart[indexCV[right]];
739 
740  decreaseParentCnt(rightID);
741  curPart[indexCV[right]] = leftID;
742  increaseParentCnt(leftID);
743 
744  changed = right;
745  } else if(isa<BinaryOperator>(&I)) {
746  left = I.getOperand(0);
747  right = I.getOperand(1);
748  IID = curPart[indexCV[&I]];
749  char op = getOpSymbol(I.getOpcodeName());
750  newID = findIDstrct(curPart, op, left, right);
751 
752  decreaseParentCnt(IID);
753  curPart[indexCV[&I]] = newID;
754  increaseParentCnt(newID);
755 
756  changed = &I;
757  } else if(isa<CallInst>(I)) {
758  IID = curPart[indexCV[&I]];
759  newID = new IDstruct;
760 
761  decreaseParentCnt(IID);
762  curPart[indexCV[&I]] = newID;
763  increaseParentCnt(newID);
764 
765  changed = &I;
766  }
767 
768  // if no variable has been changed then return,
769  // otherwise update equivalence information for
770  // every two operand expression containing the
771  // changed variable
772  if(changed == nullptr) return;
773  for(auto op : Ops)
774  for(auto x : CuV) {
775  decreaseParentCnt(curPart[indexExp[{op, x, changed}]]);
776  newID = findIDstrct(curPart, op, x, changed);
777  curPart[indexExp[{op, x, changed}]] = newID;
778  increaseParentCnt(newID);
779 
780  decreaseParentCnt(curPart[indexExp[{op, changed, x}]]);
781  newID = findIDstrct(curPart, op, changed, x);
782  curPart[indexExp[{op, changed, x}]] = newID;
783  increaseParentCnt(newID);
784  }
785  }

795  void confluenceFunction(Instruction &I, Partition &partition) {
796  vector<Instruction *> &predecessors = Predecessors[&I];
797 
798  if(predecessors.empty()) findInitialPartition(partition);
799  else {
800  // vector to check which constant/variable has already
801  // been considered while processing
802  vector<bool> accessFlag(CuV.size(), false);
803 
804  // initialize partition vector
805  partition.assign(numExps, nullptr);
806 
807  // process each constant/variable that has not
808  // already been considered
809  for(auto el : indexCV) {
810  if(not accessFlag[el.second]) {
811  accessFlag[el.second] = true;
812 
813  // check if the variable/constant is equivalent on all
814  // the predecessors, if so make it point to the same
815  // IDstruct object for the confluence partition. Notice
816  // that this case will always be true for a constant.
817  // Also another thing is if in a partition, the variable/
818  // constant contains nullptr we have to just ignore it,
819  // as theoretically it means that the corresponding partition
820  // represents the top element
821  bool flag = true;
822  IDstruct *ptr = nullptr;
823  for(int i = 0; i < (int)predecessors.size() and flag; i++) {
824  IDstruct *nptr = partitions[predecessors[i]][el.second];
825  if(nptr == nullptr) continue;
826  else if(ptr == nullptr) ptr = nptr;
827  else if(ptr != nptr) flag = false;
828  }
829 
830  if(flag) {
831  partition[el.second] = ptr;
832  increaseParentCnt(ptr);
833  } else {
834  // however if the previous case is not
835  // true, we must create a new IDstruct
836  // object to represent the class of
837  // the varible in the temporary partition.
838  // Also, every other variable that is
839  // equivalent (belong to the same class)
840  // to the current variable at all the
841  // predecessors must also point to the same
842  // IDstruct object
843  set<Value *> setCV, intersection(CuV);
844  set<expTuple> setExp;
845  for(auto inst : predecessors) {
846  getClass(el.second, partitions[inst], setCV, setExp);
847  setIntersect<Value *>(intersection, setCV);
848  }
849 
850  IDstruct *ID = new IDstruct();
851  for(auto el_ : intersection) {
852  accessFlag[indexCV[el_]] = true;
853  partition[indexCV[el_]] = ID;
854  increaseParentCnt(ID);
855  }
856  }
857  }
858  }
859 
860  // now consider every two operand expression
861  for(auto el : indexExp) {
862  Value *left = get<1>(el.first);
863  Value *right = get<2>(el.first);
864  char op = get<0>(el.first);
865 
866  // find ID struct object coressponding to the expression
867  IDstruct *ID = findIDstrct(partition, op, left, right);
868  partition[el.second] = ID;
869  increaseParentCnt(ID);
870  }
871  }
872  }

1107  partitions.clear();

1110  // one in which every element is assigned to nullptr.
1111  // This is equivalent to the top element theoretically.
